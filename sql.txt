
-- BANCO DE LA PLACETA - ESQUEMA INTEGRAL SIDS v4.5 (Capítulo 1)

-- 1. Usuarios (Ciudadanos)
CREATE TABLE IF NOT EXISTS users (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    dip TEXT UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    role TEXT DEFAULT 'CITIZEN',
    verified BOOLEAN DEFAULT FALSE,
    avatar_url TEXT,
    birth_date DATE,
    is_blocked BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 2. Cuentas Bancarias
CREATE TABLE IF NOT EXISTS accounts (
    id TEXT PRIMARY KEY,
    owner_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    alias TEXT NOT NULL,
    iban TEXT UNIQUE NOT NULL,
    balance DECIMAL(20,2) DEFAULT 0.00,
    type TEXT NOT NULL,
    company_id TEXT, -- Vinculación con sociedad
    shared_with TEXT[] DEFAULT '{}', -- Array de IDs de usuarios con acceso
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 3. Sociedades (Empresas)
CREATE TABLE IF NOT EXISTS companies (
    id TEXT PRIMARY KEY,
    owner_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    nif TEXT UNIQUE NOT NULL,
    iban TEXT UNIQUE NOT NULL,
    employees JSONB DEFAULT '[]'::jsonb,
    capital DECIMAL(20,2) DEFAULT 0.00,
    is_public BOOLEAN DEFAULT FALSE,
    stock_value DECIMAL(20,2) DEFAULT 100.00,
    price_history JSONB DEFAULT '[]'::jsonb,
    catalog JSONB DEFAULT '{"active": true, "products": []}'::jsonb,
    shareholders JSONB DEFAULT '[]'::jsonb,
    api_key TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 4. Transacciones (Ledger Central)
CREATE TABLE IF NOT EXISTS transactions (
    id TEXT PRIMARY KEY,
    amount DECIMAL(20,2) NOT NULL,
    sender_iban TEXT,
    receiver_iban TEXT NOT NULL,
    description TEXT,
    type TEXT NOT NULL,
    tax DECIMAL(20,2) DEFAULT 0.00,
    status TEXT DEFAULT 'COMPLETED',
    date TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 5. Tarjetas SIDS
CREATE TABLE IF NOT EXISTS cards (
    id TEXT PRIMARY KEY,
    account_id TEXT NOT NULL REFERENCES accounts(id) ON DELETE CASCADE,
    pan TEXT UNIQUE NOT NULL,
    expiry_date TEXT NOT NULL,
    cvv TEXT NOT NULL,
    pin TEXT NOT NULL,
    holder_name TEXT NOT NULL,
    status TEXT DEFAULT 'ACTIVE',
    design TEXT,
    format TEXT DEFAULT 'VIRTUAL',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 6. Loterías / LootBoxes
CREATE TABLE IF NOT EXISTS loot_boxes (
    id TEXT PRIMARY KEY,
    owner_id TEXT, -- NULL si es del estado
    name TEXT NOT NULL,
    description TEXT,
    price DECIMAL(20,2) NOT NULL,
    prize_value DECIMAL(20,2) NOT NULL,
    status TEXT DEFAULT 'APPROVED',
    color TEXT DEFAULT 'from-slate-700 to-slate-900',
    created_at TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP
);

-- 7. Planes de Suscripción
CREATE TABLE IF NOT EXISTS subscription_plans (
    id TEXT PRIMARY KEY,
    company_id TEXT NOT NULL REFERENCES companies(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    description TEXT,
    price DECIMAL(20,2) NOT NULL,
    billing_cycle TEXT DEFAULT 'MONTHLY',
    permanence_months INTEGER DEFAULT 0
);

-- 8. Suscripciones de Usuario
CREATE TABLE IF NOT EXISTS user_subscriptions (
    id TEXT PRIMARY KEY,
    plan_id TEXT NOT NULL,
    user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    account_id TEXT NOT NULL,
    company_id TEXT NOT NULL,
    start_date TIMESTAMP WITH TIME ZONE DEFAULT CURRENT_TIMESTAMP,
    next_payment_date TIMESTAMP WITH TIME ZONE,
    status TEXT DEFAULT 'ACTIVE',
    permanence_end_date TIMESTAMP WITH TIME ZONE
);

-- 9. Diseños de Tarjetas (Market)
CREATE TABLE IF NOT EXISTS card_designs (
    id TEXT PRIMARY KEY,
    name TEXT NOT NULL,
    image_url TEXT NOT NULL,
    rarity TEXT NOT NULL,
    price DECIMAL(20,2) NOT NULL,
    owner_id TEXT REFERENCES users(id) ON DELETE SET NULL,
    is_listed BOOLEAN DEFAULT TRUE,
    seller_name TEXT
);

-- 10. Solicitudes de Vínculo (Cuentas compartidas)
CREATE TABLE IF NOT EXISTS share_requests (
    id TEXT PRIMARY KEY,
    from_user_id TEXT NOT NULL REFERENCES users(id),
    from_user_name TEXT,
    to_user_dip TEXT NOT NULL,
    to_user_name TEXT,
    account_id TEXT NOT NULL REFERENCES accounts(id),
    access_level TEXT DEFAULT 'READ_ONLY',
    status TEXT DEFAULT 'PENDING'
);

-- 11. Contactos (Agenda)
CREATE TABLE IF NOT EXISTS contacts (
    id TEXT PRIMARY KEY,
    user_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    name TEXT NOT NULL,
    iban TEXT NOT NULL
);

-- Índices de Auditoría SIDS
CREATE INDEX idx_user_dip ON users(dip);
CREATE INDEX idx_acc_iban ON accounts(iban);
CREATE INDEX idx_tx_ibans ON transactions(sender_iban, receiver_iban);
CREATE INDEX idx_sub_user ON user_subscriptions(user_id);
